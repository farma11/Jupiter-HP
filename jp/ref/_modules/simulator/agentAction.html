[docs]class AbstractAction(metaclass=ABCMeta):
    """
    自動交渉エージェントが起こすActionの抽象クラス
    各Actionはこのクラスを継承すること
    """
    @abstractmethod
    def __init__(self, agent_id: int):
        """
        :param int agent_id: 行動を起こしたエージェントに割り振られるid
        """
        self.__bid = None
        self.__time_offered = None
